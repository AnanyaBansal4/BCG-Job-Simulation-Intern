def simple_chatbot(user_query):
    user_query1 = user_query.lower().split()
    company = None
    year1 = None
    year2 = None
    years = []
    
    # Extracting Company
    for i in user_query1:
        if i.capitalize() in df['Company'].values:
            company = i.capitalize()
            break
    
    # Extracting Years
    for i in user_query1:
        if i.isdigit() and int(i) in df['Year'].values:
            years.append(int(i))
    
    # Assign year1 and year2 if available
    if len(years) > 0:
        year1 = years[0]
        if len(years) > 1:
            year2 = years[1]

    # Total Revenue
    if "total revenue" in user_query.lower() and not "compare" in user_query.lower() and not "change" in user_query.lower():
        if company and year1:
            revenue = df.loc[(df['Company'] == company) & (df['Year'] == year1), 'Total Revenue']
            if not revenue.empty:
                print(f"Total Revenue for {company} in {year1} was {revenue.iloc[0]}")
            else:
                print(f"No revenue data found for {company} in {year1}.")
        else:
            print("Could not identify both the company and the year from the query.")

    # Total Assets
    elif "total assets" in user_query.lower():
        if company and year1:
            assets = df.loc[(df['Company'] == company) & (df['Year'] == year1), 'Total Assets']
            if not assets.empty:
                print(f"Total Assets for {company} in {year1} was {assets.iloc[0]}")
            else:
                print(f"No data found for {company} in {year1}.")
        else:
            print("Could not identify both the company and the year from the query.")

    # Total Liabilities
    elif "total liabilities" in user_query.lower():
        if company and year1:
            liabilities = df.loc[(df['Company'] == company) & (df['Year'] == year1), 'Total Liabilities']
            if not liabilities.empty:
                print(f"Total Liabilities for {company} in {year1} was {liabilities.iloc[0]}")
            else:
                print(f"No data found for {company} in {year1}.")
        else:
            print("Could not identify both the company and the year from the query.")

    # Cash Flow
    elif "cash flow" in user_query.lower():
        if company and year1:
            cash_flow = df.loc[(df['Company'] == company) & (df['Year'] == year1), 'Cash Flow']
            if not cash_flow.empty:
                print(f"Cash Flow for {company} in {year1} was {cash_flow.iloc[0]}")
            else:
                print(f"No data found for {company} in {year1}.")
        else:
            print("Could not identify both the company and the year from the query.")

    # Net Income Change
    elif "net income" in user_query.lower() and not "compare" in user_query.lower() and not "change" in user_query.lower():
        if company and year1:
            net_income = df.loc[(df['Company'] == company) & (df['Year'] == year1), 'Net Income']
            if not net_income.empty:
                print(f"Net Income for {company} in {year1} was {net_income.iloc[0]}")
            else:
                print(f"No data found for {company} in {year1}.")
        else:
            print("Could not identify both the company and the year from the query.")

    # Change in Net Income
    elif "net income" in user_query.lower() and "change" in user_query.lower():
        if company and year1 and year2:
            net_income_year1 = df.loc[(df['Company'] == company) & (df['Year'] == year1), 'Net Income'].values
            net_income_year2 = df.loc[(df['Company'] == company) & (df['Year'] == year2), 'Net Income'].values
            if net_income_year1.size > 0 and net_income_year2.size > 0:
                change = net_income_year2[0] - net_income_year1[0]
                print(f"The change in net income for {company} from {year1} to {year2} is {change}")
            else:
                print(f"Could not find net income data for {company} in {year1} or {year2}.")
        else:
            print("Could not identify both the company and the years from the query.")

                print(f"The average revenue growth for {company} is {growth_rate:.2f}%")
            else:
                print(f"No revenue growth data found for {company}.")
        else:
            print("Could not identify the company from the query.")
